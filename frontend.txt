  late Socket socket;

  void connect(String token) {
    socket = io(
      'wss://api.cysy.social',
      OptionBuilder()
        .setTransports(['websocket'])
        .setExtraHeaders({'Authorization': 'Bearer $token'})
        .enableAutoConnect()
        .enableReconnection()
        .build(),
